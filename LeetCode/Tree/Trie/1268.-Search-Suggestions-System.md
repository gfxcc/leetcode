	class Node:  
	    def __init__(self):  
	        self.words = []  
	        self.next = defaultdict(Node)  
	class Solution:  
	    def suggestedProducts(self, products: List[str], searchWord: str) -> List[List[str]]:  
	        products = sorted(products)  
	        tries = Node()  
	        for product in products:  
	            cur = tries  
	            for c in product:  
	                cur = cur.next[c]  
	                cur.words.append(product)  
	        ret, cur = [], tries  
	        for c in searchWord:  
	            cur = cur.next[c]  
	            ret.append(cur.words[:3])  
	        return ret
