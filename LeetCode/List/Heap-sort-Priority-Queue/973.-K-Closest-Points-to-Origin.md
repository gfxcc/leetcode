	class Solution:  
	    def kClosest(self, points: List[List[int]], K: int) -> List[List[int]]:  
	        pq = []  
	        for x, y in points:  
	            dis = (x*x + y*y)  
	            heapq.heappush(pq, (-dis, (x, y)))  
	            if len(pq) > K:  
	                heapq.heappop(pq)  
	        return [list(point) for _, point in pq]
