	class Solution:  
	    def addBinary(self, a: str, b: str) -> str:  
	        a, b = list(a), list(b)  
	        carry, rst = False, ''  
	        while a or b:  
	            v1, v2 = int(a.pop()) if a else 0, int(b.pop()) if b else 0  
	            v = v1 + v2 + carry  
	            rst += str(v % 2)  
	            carry = v // 2  
	        if carry:  
	            rst += '1'  
	        return rst[::-1]
