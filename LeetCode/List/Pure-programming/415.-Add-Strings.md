	```python  
	class Solution:  
	    def addStrings(self, num1: str, num2: str) -> str:  
	        rst = ''  
	        carry = False  
	        num1, num2 = list(num1), list(num2)  
	        while num1 or num2:  
	            cur = (int(num1.pop()) if num1 else 0) + (int(num2.pop()) if num2 else 0) + carry  
	            rst += str(cur % 10)  
	            carry = cur // 10  
	        if carry:  
	            rst += '1'  
	        return rst[::-1]  
	```
