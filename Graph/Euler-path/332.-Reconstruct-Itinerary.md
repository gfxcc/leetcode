```python  
class Solution:  
    def findItinerary(self, tickets: List[List[str]]) -> List[str]:  
        graph = defaultdict(list)  
        for a, b in sorted(tickets)[::-1]:  
            graph[a].append(b)  
          
        ret = []  
        def dfs(cur):  
            while graph[cur]:  
                nxt = graph[cur].pop()  
                dfs(nxt)  
            ret.append(cur)  
        dfs('JFK')  
        return ret[::-1]  
```
