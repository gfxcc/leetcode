class Solution:  
    def findOrder(self, N: int, P: List[List[int]]) -> List[int]:  
        graph = defaultdict(list)  
        degree = Counter()  
        for a, b in P:  
            degree[a] += 1  
            graph[b].append(a)  
        q = deque([i for i in range(N) if degree[i] == 0])  
        ret = []  
        while q:  
            c = q.popleft()  
            ret.append(c)  
            for nxt in graph[c]:  
                degree[nxt] -= 1  
                if degree[nxt] == 0:  
                    q.append(nxt)  
        return ret if len(ret) == N else []
