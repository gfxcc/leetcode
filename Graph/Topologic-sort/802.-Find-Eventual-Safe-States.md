class Solution:  
    def eventualSafeNodes(self, POINTS: List[List[int]]) -> List[int]:  
        graph = defaultdict(set)  
        degree = Counter()  
        for u, nxt in enumerate(POINTS):  
            for v in nxt:  
                graph[v].add(u)  
            degree[u] = len(nxt)  
          
        q = [i for i in range(len(POINTS)) if degree[i] == 0]  
        ret = []  
        while q:  
            u = q.pop()  
            ret.append(u)  
            for v in graph[u]:  
                degree[v] -= 1  
                if degree[v] == 0:  
                    q.append(v)  
        return sorted(ret)
