class Solution:  
    def canVisitAllRooms(self, rooms: List[List[int]]) -> bool:  
        q = [0]  
        visited = set()  
        while q:  
            u = q.pop()  
            visited.add(u)  
            for v in rooms[u]:  
                if v not in visited:  
                    q.append(v)  
          
        return len(visited) == len(rooms)
