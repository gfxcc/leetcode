class Solution:  
    def numSquares(self, n: int) -> int:  
        if n <= 2: return n  
        square_table = [i*i for i in range(1, int(n**0.5)+1)]  
        q, nq = [n], []  
        rst = 0  
        while q:  
            rst += 1  
            for x in q:  
                for i in square_table:  
                    if i == x:  
                        return rst  
                    elif i > x:  
                        break  
                    nq.append(x-i)  
            q, nq = nq, []  
        return rst
