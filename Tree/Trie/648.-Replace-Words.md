class Trie:  
    def __init__(self):  
        T = lambda: defaultdict(T)  
        self.root = T()  
      
    def add(self, word):  
        cur = self.root  
        for c in word:  
            cur = cur[c]  
        cur['$'] = word  

    def replace(self, word):  
        cur = self.root  
        for c in word:  
            if '$' in cur: return cur['$']  
            if c not in cur: return word  
            cur = cur[c]  
        return word  
          
class Solution:  
    def replaceWords(self, words: List[str], sentence: str) -> str:  
        trie = Trie()  
        for w in words:  
            trie.add(w)  
        return ' '.join(trie.replace(word) for word in sentence.split())
