class TreeNode:  
    def __init__(self):  
        self.words = []  
        self.next = defaultdict(TreeNode)  
          
class Trie:  
    def __init__(self):  
        self.root = TreeNode()  
      
    def add(self, word):  
        cur = self.root  
        for c in word:  
            cur = cur.next[c]  
            cur.words.append(word)  
          
    def search(self, word):  
        cur = self.root  
        for c in word:  
            cur = cur.next[c]  
        return cur.words  

class Solution:  
    def wordSquares(self, words: List[str]) -> List[List[str]]:  
        def dfs(seq):  
            if len(seq) == len(seq[0]):   
                ret.append(list(seq))  
                return  
            idx = len(seq)  
            for cand in trie.search(''.join(row[idx] for row in seq)):  
                seq.append(cand)  
                dfs(seq)  
                seq.pop()  
                  
        trie, ret = Trie(), []  
        for w in words:  
            trie.add(w)  
          
        for i, w in enumerate(words):  
            seq = [w]  
            dfs(seq)  
        return ret
