	class Solution:  
	    def generateParenthesis(self, n: int) -> List[str]:  
	        def dfs(l, r, s):  
	            if not l and not r:  
	                rst.append(s)  
	                return  
	            if l:  
	                dfs(l-1, r, s+'(')  
	            if l < r:  
	                dfs(l, r-1, s+')')  
	        rst = []  
	        dfs(n, n, '')  
	        return rst
