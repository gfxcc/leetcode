class Solution:  
    def numSquarefulPerms(self, A: List[int]) -> int:  
        c = Counter(A)  
        cand = {x: [y for y in c if (int((x+y)**0.5))**2 == (x+y)] for x in c}  
        print(cand)  
        def dfs(x, size=1):  
            if size == len(A): return 1  
            c[x] -= 1  
            ret = sum(dfs(y, size+1) for y in cand[x] if c[y])  
            c[x] += 1  
            return ret  
        return sum(map(dfs, c))
