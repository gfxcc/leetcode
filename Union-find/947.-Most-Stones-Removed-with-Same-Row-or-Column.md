class Solution:  
    def removeStones(self, stones: List[List[int]]) -> int:  
        uf = {}  
        def find(x):  
            if x != uf[x]:  
                uf[x] = find(uf[x])  
            return uf[x]  
        def union(a, b):  
            uf.setdefault(a, a)  
            uf.setdefault(b, b)  
            uf[find(a)] = uf[find(b)]  
          
        for x, y in stones:  
            union(x, ~y)  
        return len(stones) - len(set(map(find , uf)))
