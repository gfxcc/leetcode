
	Guess answer by using binary search  
	
```python  
class Solution:  
    def splitArray(self, nums: List[int], m: int) -> int:  
        def valid_sum(max_sum):  
            cnt, cur = 1, 0  
            for num in nums:  
                cur += num  
                if cur > max_sum:  
                    cnt += 1  
                    cur = num  
                    if cnt > m:  
                        return False  
            return True  
        # Binary search  
        l, r = max(nums), sum(nums)  
        while l < r:  
            mid = (l+r) // 2  
            if valid_sum(mid):  
                r = mid  
            else:  
                l = mid+1  
        return l  
```
