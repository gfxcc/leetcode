class Solution:  
    def removeDuplicateLetters(self, s: str) -> str:  
        ret, start = '', 0  
        chars = set(s)  
        while chars:  
            for c in string.ascii_lowercase:  
                if c in ret or c not in s: continue  
                if set(s[s.index(c):]) >= chars:  
                    ret += c  
                    chars.remove(c)  
                    s = s[s.index(c)+1:]  
                    break  
        return ret
