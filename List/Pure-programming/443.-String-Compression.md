class Solution:  
    def compress(self, chars: List[str]) -> int:  
        idx1, idx2, cnt = 0, 0, 0  
        chars.append('0')  
        for i, c in enumerate(chars):  
            if i != 0 and chars[i] != chars[i-1]:  
                chars[idx1] = chars[i-1]  
                idx1 += 1  
                if cnt == 1:  
                    continue  
                for cc in str(cnt):  
                    chars[idx1] = cc  
                    idx1 += 1  
                cnt = 1  
            else:  
                cnt += 1  
                  
        return idx1
