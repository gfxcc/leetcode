	class Solution:  
	    def findAndReplacePattern(self, words: List[str], pattern: str) -> List[str]:  
	        def is_valid(word):  
	            if len(word) != len(pattern):  
	                return False  
	            d_12, d_21 = {}, {}  
	            for c1, c2 in zip(word, pattern):  
	                if c1 not in d_12:  
	                    if c2 in d_21:  
	                        return False  
	                    d_12[c1] = c2  
	                    d_21[c2] = c1  
	                elif d_12[c1] != c2:  
	                    return False  
	            return True  
	        rst = []  
	        for word in words:  
	            if is_valid(word):  
	                rst.append(word)  
	        return rst
