```python  
class Solution:  
    def groupStrings(self, strings: List[str]) -> List[List[str]]:  
          
        d = defaultdict(list)  
        for s in strings:  
            delta = ord(s[0]) - ord('a')  
            ss = ''.join(chr(97+((ord(c)-97-delta) % 26)) for c in s)  
            d[ss].append(s)  
        return d.values()  
```
