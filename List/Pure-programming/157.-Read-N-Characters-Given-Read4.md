class Solution:  
    def read(self, buf, n):  
        """  
        :type buf: Destination buffer (List[str])  
        :type n: Number of characters to read (int)  
        :rtype: The number of actual characters read (int)  
        """  
        if n <= 0: return 0  
        for i in range(n)[::4]:  
            tmp = [0] * 4  
            size = read4(tmp)  
            for j in range(size):  
                buf[i+j] = tmp[j]  
            if size != 4: break  
        return min(n, i+size)
