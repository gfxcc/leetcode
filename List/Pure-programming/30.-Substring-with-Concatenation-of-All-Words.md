	class Solution:  
	    def findSubstring(self, s: str, words: List[str]) -> List[int]:  
	        if not words:  
	            return []  
	        c = Counter(words)  
	        size = len(words[0])  
	        rst = []  
	        for i in range(len(s)+1-size*len(words)):  
	            j, cc = i, c.copy()  
	            while j < i+size*len(words) and s[j:j+size] in c:  
	                cur_w = s[j:j+size]  
	                cc[cur_w] -= 1  
	                if cc[cur_w] < 0:  
	                    break  
	                j += size  
	            if all(v == 0 for v in cc.values()):  
	                rst.append(i)  
	        return rst
