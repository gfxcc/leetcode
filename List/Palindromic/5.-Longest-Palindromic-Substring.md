	```python  
	class Solution:  
	    def longestPalindrome(self, s: str) -> str:  
	        ret, i = '', 0  
	          
	        while i < len(s):  
	            j = i  
	            while j < len(s) and s[j] == s[i]:  
	                j += 1  
	              
	            k = 0  
	            while i-k >= 0 and j+k-1 < len(s) and s[i-k] == s[j+k-1]:  
	                if j-i+2*k > len(ret):  
	                    ret = s[i-k:j+k]  
	                k += 1  
	              
	            i = j  
	        return ret            
	```
