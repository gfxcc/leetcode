	class Solution:  
	    def validateStackSequences(self, pushed: List[int], popped: List[int]) -> bool:  
	        dq = deque(pushed)  
	        stack = []  
	        for val in popped:  
	            if stack and stack[-1] == val:  
	                stack.pop()  
	            else:  
	                while dq:  
	                    stack.append(dq.popleft())  
	                    if stack[-1] == val:  
	                        break  
	                if stack[-1] != val:  
	                    return False  
	                stack.pop()  
	        return not dq and not stack
