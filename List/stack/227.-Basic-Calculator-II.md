```python  
class Solution:  
    def calculate(self, s: str) -> int:  
        s = s.strip().replace(' ', '')  
        nums = []  
        sign, val = '+', 0  
        for i, c in enumerate(s):  
            if c.isdigit():  
                val = val*10 + int(c)  
            if not c.isdigit() or i == len(s)-1:  
                if sign == '+':  
                    nums.append(val)  
                elif sign == '-':  
                    nums.append(-val)  
                elif sign == '*':  
                    nums.append(nums.pop() * val)  
                elif sign == '/':  
                    nums.append(int(nums.pop() / val))  
                sign = c  
                val = 0  
        return sum(nums)  
```
