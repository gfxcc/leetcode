	class Solution:  
	    def lengthLongestPath(self, input: str) -> int:  
	        ret, stack, cur = 0, [], 0  
	          
	        for line in input.split('\n'):  
	            depth = line.count('\t')  
	            pattern = line.lstrip('\t')  
	              
	            while depth < len(stack):  
	                cur -= len(stack.pop()) + 1  
	              
	            if '.' in line:  
	                ret = max(ret, cur + len(pattern))  
	            else:  
	                stack.append(pattern)  
	                cur += 1 + len(pattern)  
	              
	        return ret
