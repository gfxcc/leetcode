
```python  
class Solution:  
    def isValid(self, s: str) -> bool:  
        stack = []  
        pairs = {'(': ')', '[': ']', '{': '}'}  
        for c in s:  
            if c in ['(', '[', '{']:  
                stack.append(pairs[c])  
            else:  
                if stack and stack[-1] == c:  
                    stack.pop()  
                else:  
                    return False  
        return not stack  
```
