class Solution:  
    def numberWays(self, hats: List[List[int]]) -> int:  
        h2p = defaultdict(list)  
        for i, hat in enumerate(hats):  
            for h in hat:  
                h2p[h].append(i)  
        n = len(hats)  

        @lru_cache(None)  
        def dp(hat, mask):  
            if mask == 0: return 1  
            if hat < 1: return 0  
              
            ret = dp(hat-1, mask)  
            for p in h2p[hat]:  
                if mask & (1 << p):  
                    mask ^= (1 << p)  
                    ret += dp(hat-1, mask)  
                    mask |= (1 << p)  
            return ret % (10**9 + 7)  
          
        return dp(40, (1<<n)-1)
