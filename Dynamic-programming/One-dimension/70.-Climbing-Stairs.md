	class Solution:  
	    def climbStairs(self, n: int) -> int:  
	        '''          
	        d = {0:1, 1:1}  
	        def helper(k):  
	            if k in d: return d[k]  
	            d[k] = helper(k-1) + helper(k-2)  
	            return d[k]  
	        return helper(n)  
	        '''  
	        a, b = 1, 1  
	        if n == 1: return 1  
	        while n > 1:  
	            a, b = b, a+b  
	            n-=1  
	        return b
