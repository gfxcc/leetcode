	```python  
	class Solution:  
	    def rob(self, nums: List[int]) -> int:  
	        if not nums: return 0  
	        '''  
	        # bottom up  
	        a, b = 0, nums[0]  
	        for v in nums[1:]:  
	            a, b = b, max(b, a+v)  
	        return b  
	        '''  
	        # Top down  
	        d = {-1: 0, 0: nums[0]}  
	        def helper(k):  
	            if k in d: return d[k]  
	            d[k] = max(helper(k-2)+nums[k], helper(k-1))  
	            return d[k]  
	        return helper(len(nums)-1)  
	```
