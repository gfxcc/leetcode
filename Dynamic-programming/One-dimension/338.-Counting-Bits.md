	```python  
	class Solution:  
	    def countBits(self, num: int) -> List[int]:  
	        dp = [0] * (num+1)  
	        if num == 0: return dp  
	        dp[1], prev = 1, 1  
	        for i in range(2, num+1):  
	            if i == prev * 2:  
	                dp[i] = 1  
	                prev = i  
	            else:  
	                dp[i] = 1 + dp[i-prev]  
	        return dp  
	```
